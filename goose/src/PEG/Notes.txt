type identifier = [A-Za-z][A-Za-z_0-9]*

data peg_expression = 
        (* choice peg_expression peg_expression *)
    |   primitive string
    |   non-terminal identifier
    |   sequence peg_expression peg_expression
    |   not peg_expression
    |   empty
    |   repeat peg_expression

data parse_tree =
        non-term identifier * int * list parse_tree
    |   terminal string

span : parse_tree -> int

data parse_result = 
    | success parse_tree
    | fail

parse_expression : peg_expression -> string -> parse_tree

parse : peg -> string -> parse_tree


VC <- LoopVariant / LoopInvariant / Writes / Requires / Ensures / Valid / Separated
// Expression <- PredicateCall / Quantifier / Infix_Expression(Atom, Infix_Operator)
Expression <- PredicateCall / Quantifier / TEST
TEST <- Atom (Infix_Operator Atom)*
Atom       <- '(' Expression ')' / Literal / WriteableExpression / Prefix_Operator Atom
PredicateCall  <- VariableReference '(' ExpressionList ')'
ExpressionList <- Expression (',' Expression)*
Quantifier <- Quant VarList '.' Expression
Quant      <- 'forall' / 'exists'
VarList    <- VariableReference (',' VariableReference)*
Literal <- Number
WriteableExpression <- VariableReference / PointerDereference
VariableReference   <- Identifier
PointerDereference  <- '*' WriteableExpression
WriteableExpressionList <- WriteableExpression (',' WriteableExpression)*
LoopVariant   <- 'variant' Expression
LoopInvariant <- 'invariant' Expression
Writes        <- 'writes' WriteableExpressionList
Requires      <- 'requires' Expression
Ensures       <- 'ensures' Expression
Valid         <- 'valid' WriteableExpressionList
Separated     <- 'separated' WriteableExpressionList
Identifier <- < !Keyword Name >
Keyword    <- ('exists' | 'forall' | 'variant' | 'invariant' | 'writes' | 'requires' | 'ensures' | 'valid' | 'separated') !Name
Name   <- < [A-Za-z_][A-Za-z0-9_]* >
Number <- < [0-9]+ >
Prefix_Operator <- < [!-] >
Infix_Operator  <- < '->' / '<->' / '>=' / '<=' / '\\/' / '/\\' / '==' / '!=' / [-+/><*] >
%whitespace <- [ \t]*
%word       <- [A-Za-z]+
Infix_Expression(A, O) <- A (O A)* {
  precedence
    L -> <->
    L /\\ \\/
    L >= <= > < == !=
    L + -
    L * /
